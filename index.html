function doGet(e) { var action = e.parameter.action; var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('Trades') || ss.insertSheet('Trades'); if (action === 'getAll') { var data = sheet.getDataRange().getValues(); if (data.length <= 1) { return ContentService.createTextOutput(JSON.stringify({status: 'success', trades: []})) .setMimeType(ContentService.MimeType.JSON); } var trades = []; for (var i = 1; i < data.length; i++) { var row = data[i]; trades.push({ id: row[21] || '', date: row[0] || '', time: row[1] || '', pair: row[2] || '', dir: row[3] || '', lot: row[4] || 0, sl: row[5] || 0, tp: row[6] || 0, rr: row[7] || 0, session: row[8] || '', setup: row[9] || '', duration: row[10] || '', type: row[11] || '', result: row[12] || 0, balanceBefore: row[13] || 0, balanceAfter: row[14] || 0, dailyLoss: row[15] || 0, firm: row[16] || '', psychology: row[17] || '', note: row[18] || '', rating: row[19] || '', image: row[20] || '' }); } return ContentService.createTextOutput(JSON.stringify({status: 'success', trades: trades})) .setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { try { var data = JSON.parse(e.postData.contents); var action = data.action; var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('Trades') || ss.insertSheet('Trades'); if (sheet.getLastRow() === 0) { sheet.appendRow(['Date', 'Time', 'Pair', 'Direction', 'Lot', 'SL', 'TP', 'RR', 'Session', 'Setup', 'Duration', 'Type', 'Result', 'Balance Before', 'Balance After', 'Daily Loss', 'Firm', 'Psychology', 'Notes', 'Rating', 'Screenshot', 'ID']); sheet.getRange('A1:V1').setFontWeight('bold').setBackground('#0c1220').setFontColor('#00e5ff'); sheet.setFrozenRows(1); } if (action === 'add') { var t = data.trade; var imageUrl = ""; // ছবি গুগল ড্রাইভে আপলোড if (t.imageBase64) { try { var folder = DriveApp.getFoldersByName("TradeScreenshots"); if (!folder.hasNext()) folder = DriveApp.createFolder("TradeScreenshots"); folder = folder.next(); var decoded = Utilities.base64Decode(t.imageBase64); var blob = Utilities.newBlob(decoded, 'image/png', t.id + '.png'); var file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); imageUrl = file.getUrl(); } catch (imgErr) { Logger.log(imgErr.toString()); } } sheet.appendRow([ t.date || '', t.time || '', t.pair || '', t.dir || '', t.lot || 0, t.sl || 0, t.tp || 0, t.rr || 0, t.session || '', t.setup || '', t.duration || '', t.type || '', t.result || 0, t.balanceBefore || 0, t.balanceAfter || 0, t.dailyLoss || 0, t.firm || '', t.psychology || '', t.note || '', t.rating || '', imageUrl, t.id || '' ]); createDashboard_(ss); return ContentService.createTextOutput(JSON.stringify({status: 'success', message: 'Trade added'})) .setMimeType(ContentService.MimeType.JSON); } } catch (err) { return ContentService.createTextOutput(JSON.stringify({status: 'error', message: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function createDashboard_(ss) { var dash = ss.getSheetByName('Dashboard') || ss.insertSheet('Dashboard', 0); dash.clear(); dash.getRange('A1').setValue('Professional Trading Dashboard').setFontWeight('bold').setFontSize(16).setFontColor('#00e5ff'); dash.getRange('A3').setValue('Summary & Equity Curve').setFontWeight('bold').setFontColor('#ffab00'); var stats = [ ['Total Trades', '=COUNTA(Trades!C2:C)'], ['Win Rate (%)', '=IFERROR(COUNTIF(Trades!L2:L, "win")/COUNTA(Trades!L2:L)*100, 0)'], ['Net P&L ($)', '=SUM(Trades!M2:M)'] ]; dash.getRange('A4:B6').setValues(stats).setFontFamily('Courier New'); dash.getRange('A4:A6').setBackground('#111827').setFontColor('#94a3b8'); dash.getRange('B4:B6').setFontWeight('bold'); dash.getRange('D3').setValue('Equity Curve').setFontWeight('bold').setFontColor('#ffab00'); dash.getRange('D4').setFormula('=SPARKLINE(Trades!N2:N, {"charttype","line"; "color","#00e676"; "linewidth",2})'); dash.setRowHeight(4, 60); dash.getRange('A8').setValue('Day of Week Performance').setFontWeight('bold').setFontColor('#ffab00'); dash.getRange('A9:C9').setValues([['Day', 'Trades', 'Net P&L']]).setFontWeight('bold').setBackground('#111827'); var dowFormula = '=QUERY(Trades!A1:M, "SELECT TEXT(A, \'dddd\'), COUNT(A), SUM(M) WHERE A IS NOT NULL GROUP BY TEXT(A, \'dddd\') LABEL TEXT(A, \'dddd\') \'Day\', COUNT(A) \'Trades\', SUM(M) \'P&L\'", 0)'; dash.getRange('A10').setFormula(dowFormula); dash.getRange('E8').setValue('Setup Analysis').setFontWeight('bold').setFontColor('#ffab00'); dash.getRange('E9:G9').setValues([['Setup', 'Trades', 'Net P&L']]).setFontWeight('bold').setBackground('#111827'); var setupFormula = '=QUERY(Trades!J1:M, "SELECT J, COUNT(J), SUM(M) WHERE J IS NOT NULL GROUP BY J LABEL J \'Setup\', COUNT(J) \'Trades\', SUM(M) \'P&L\'", 0)'; dash.getRange('E10').setFormula(setupFormula); }
